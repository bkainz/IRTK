PROJECT(RVIEW_GLUT)

LINK_LIBRARIES(rview++ registration++ transformation++ image++ geometry++
common++) 

INCLUDE (${CMAKE_ROOT}/Modules/FindGLUT.cmake)

 find_package(GLUT REQUIRED) 

IF (GLUT_FOUND)
  INCLUDE_DIRECTORIES(${GLUT_INCLUDE_PATH} ${GLUT_INCLUDE_DIR})
ENDIF(GLUT_FOUND)

IF (GLUT_FOUND)
  LINK_LIBRARIES (${GLUT_LIBRARY} ${GLUT_LIBRARIES})
  LINK_DIRECTORIES(${GLUT_INCLUDE_DIR}/../lib)
ENDIF (GLUT_FOUND)

INCLUDE (${CMAKE_ROOT}/Modules/FindOpenGL.cmake)

IF (GLU_LIBRARY)
   LINK_LIBRARIES (${GLU_LIBRARY})
ENDIF (GLU_LIBRARY)
 
IF (OPENGL_INCLUDE_PATH)
  INCLUDE_DIRECTORIES(${OPENGL_INCLUDE_PATH})
ENDIF(OPENGL_INCLUDE_PATH)

IF (OPENGL_LIBRARY)
  LINK_LIBRARIES (${OPENGL_LIBRARY})
ENDIF (OPENGL_LIBRARY)

SET(GLUT_SRCS main.cc)

ADD_EXECUTABLE(display ${GLUT_SRCS})
INSTALL_TARGETS(/bin display)
